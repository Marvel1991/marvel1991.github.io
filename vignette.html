3. Fragment size distribution of each group

#plot the fragment size distribution of each group 

#define a list
grp_list<-list("healthy"="healthy","renal"="renal","bladder"="bladder","melanoma"="melanoma")

#generating the plots and store them in a list
result<-sapply(grp_list, function(x){
  result <-callSize(path = root_path) %>% filter(group==as.character(x)) %>% plotSinglegroup()
},simplify = F)

#multiplexing the plots in one figure
#please mind that the plots were modified by adding different elements by "+"
multiplex<-ggpubr::ggarrange(result$healthy$prop_plot+theme(axis.title.x = element_blank()),

4. Median size metrics

compare_grps<-callMetrics(root_path) %>% plotMetrics(order=order)

#modify the plots
p1<-compare_grps$median_prop_plot +
  theme(axis.title.x = element_blank(),
        axis.title.y = element_text(size=12,face="bold")) +
  theme(legend.position = c(0.7, 0.5),
        legend.text = element_text( size = 11),
        legend.title = element_blank())

p2<-compare_grps$median_cdf_plot +
  scale_y_continuous(labels = scales::number_format(accuracy = 0.001)) +
  theme(axis.title=element_text(size=12,face="bold")) +
  theme(legend.position = c(0.7, 0.5),
        legend.text = element_text( size = 11),
        legend.title = element_blank())

#Finalize the plots
median_grps<-ggpubr::ggarrange(p1,
                       p2,
                       label.x = 0.3,
                       ncol = 1,
                       nrow = 2
                       )
